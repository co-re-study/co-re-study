### 내 풀이

- 방법 : 큐 중 합이 더 큰 곳에서 빼서 옮긴다.
- 시간복잡도 :

```python
from collections import deque

def solution(queue1, queue2):
    a = deque(queue1)
    b = deque(queue2)
    sum_a = sum(a)
    sum_b = sum(b)

    if (sum_a + sum_b) % 2 != 0:
        return -1

    limit = len(a)*4
    answer = 0
    while sum_a != sum_b:
        if sum_a > sum_b:
            temp = a.popleft()
            b.append(temp)
            sum_a -= temp
            sum_b += temp
            answer += 1
        else:
            temp = b.popleft()
            a.append(temp)
            sum_b -= temp
            sum_a += temp
            answer += 1

        if answer > limit:
            return -1

    return answer
```
